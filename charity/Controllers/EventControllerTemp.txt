        public async Task<IActionResult> EventLocations(int id)
        {
            Event? e = await _context.Events.FindAsync(id);
            
            if (e == null)
            {
                return NotFound();
            }

            return PartialView("_EventLocationsPartial", e.EventLocations);
        }
